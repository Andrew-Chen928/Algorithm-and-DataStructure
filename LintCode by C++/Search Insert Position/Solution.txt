class Solution {
    /** 
     * param A : an integer sorted array
     * param target :  an integer to be inserted
     * return : an integer
     */
public:
    int searchInsert(vector<int> &A, int target) {
        // write your code here
        int left = 0; 
        int right = A.size() - 1;
        while( left <= right)
        {
            int mid = left + (right - left) / 2;
            if(A[mid] >= target) 
            {   
                right = mid -1;
            }
            else 
            {
                left = mid + 1;
            }
        }
        return left;
    }
};

class Solution {
public:
    /**
     * @param digits a number represented as an array of digits
     * @return the result
     */
     
    void helper(vector<int>& digits, int value, int position)
    {
        
        if(value == 9)
        {
            digits[position] = 0;
            position--;
            helper(digits, digits[position], position);
        }
        else
        {
            digits[position]++;
        }
    }
     
    vector<int> plusOne(vector<int>& digits) {
        // Write your code here
        digits.emplace_back(0);
        rotate(digits.begin(), digits.end()-1, digits.end());
        
        int len = digits.size();
        helper(digits, digits[len-1], len-1 );
        if(digits[0] == 0) 
        {
            digits.erase(digits.begin());
        }
        return digits;
    }
};

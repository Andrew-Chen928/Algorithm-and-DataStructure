class Solution {
public:
    /**
     * @param num: a rotated sorted array
     * @return: the minimum number in the array
     */
    int findMin(vector<int> &num) {
        int start = 0, end = num.size() -1;
        while(start < end)
        {
            int mid = start + (end - start) / 2;
            if(num[mid] < num[end])
            {
                end = mid;
            }
            else
            {
                start = mid + 1;
            }
        }
        return num[start];
    }
};
